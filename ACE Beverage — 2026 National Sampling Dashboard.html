🔌 Connect Your Google Sheet
A one-time 3-minute setup that permanently fixes the connection. This works by adding a small script to your Google Sheet that serves the data directly to this dashboard.
1
Open your Google Sheet's Script Editor
Go to your Google Sheet → click Extensions in the top menu → click Apps Script
2
Replace ALL existing code with this script
Delete everything in the editor, then paste this:
function doGet(e) { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheets()[0]; var sheets = ss.getSheets(); for (var i = 0; i < sheets.length; i++) { var n = sheets[i].getName().toLowerCase(); if (n.includes('form') || n.includes('response')) { sheet = sheets[i]; break; } } var data = sheet.getDataRange().getValues(); var headers = data[0].map(String); var rows = []; for (var r = 1; r < data.length; r++) { var row = {}; for (var c = 0; c < headers.length; c++) { var v = data[r][c]; if (v instanceof Date) v = v.toISOString(); row[headers[c]] = (v === undefined || v === null) ? '' : v; } rows.push(row); } var json = JSON.stringify({headers: headers, rows: rows}); var cb = e && e.parameter && e.parameter.callback; if (cb) { return ContentService .createTextOutput(cb + '(' + json + ')') .setMimeType(ContentService.MimeType.JAVASCRIPT); } return ContentService .createTextOutput(json) .setMimeType(ContentService.MimeType.JSON); }
3
Deploy as a Web App
Click Deploy → New deployment → click the gear icon ⚙️ next to "Type" → select Web app

Set Execute as: Me
Set Who has access: Anyone

Click Deploy → authorize when prompted → copy the Web app URL
4
Paste your Web App URL below
